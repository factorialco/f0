# Colors

Factorial One uses color tokens to define colors.

export function ColorToken({ name, description }) {
  const [copied, setCopied] = useState(false)

const copyToClipboard = () => { navigator.clipboard.writeText(name);
setCopied(true); setTimeout(() => setCopied(false), 1000); };

return (

<div className="bg-f1-background-secondary border-f1-border flex max-w-[600px] flex-row items-center space-x-3 rounded-lg border border-solid px-2 py-1.5">
  <div className={cn("h-8 w-12 rounded-lg", `bg-${name}`)} />
  <div className="flex flex-1 flex-col space-y-0">
    <span className="font-mono text-f1-foreground flex-grow font-semibold">
      {name}
    </span>
    <p className="text-f1-foreground-secondary text-sm">{description}</p>
  </div>
  <Button
    onClick={copyToClipboard}
    variant="outline"
    label={copied ? "Copied" : "Copy"}
  />
</div>
) }
