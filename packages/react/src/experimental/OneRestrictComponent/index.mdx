# RestrictComponent

Restrict a component to be rendered only in certain routes.

```tsx
import { OneRestrictComponent } from "@factorialco/factorial-one/dist/experimental"
```

## Examples

### Allowed routes

```tsx
<OneRestrictComponent
  identifier="demo-component"
  allowedRoutes={["/dashboard", "/profile"]}
>
  <DemoComponent />
</OneRestrictComponent>
```

### Disallowed routes

```tsx
<OneRestrictComponent
  identifier="demo-component"
  disallowedRoutes={["/login", "/signup"]}
>
  <DemoComponent />
</OneRestrictComponent>
```

### Allowed and disallowed routes

```tsx
<OneRestrictComponent
  identifier="demo-component"
  allowedRoutes={["/dashboard", "/profile"]}
  disallowedRoutes={["/login", "/signup"]}
>
  <DemoComponent />
</OneRestrictComponent>
```
